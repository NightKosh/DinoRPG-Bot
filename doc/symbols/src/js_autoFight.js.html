152 	/** Получает время у текущего диноза */
153 	var getTime = function () {
154 		var time = $("#timer_0");
155 		if (time.length != 0) {
156 			time = time.text();
157 			var minutes = parseInt(time.slice(0, time.indexOf('m')));
158 			if (minutes == -1) {
159 				minutes = 0;
160 				
161 				var seconds = parseInt(time);
162 			} else {
163 				var seconds = parseInt(time.slice(time.indexOf('m')+ 1, time.indexOf('s')));
164 			}
165 			
166 			return minutes * 60000 + seconds * 1000 + new Date().getTime();
167 		} else {
168 			return new Date().getTime();
169 		}
170 	}

172 	/** Получает новое время */
173 	var getNewTime = function () {
174 		//если мы на нужном дине
175 		if (document.location.pathname == "/dino/" + dinozList[settings.lastCheck].id) {
176 			var time = getTime();
177 			if (time > settings.time) {
178 				saveSettings("time", time);
179 			}
180 			
181 			if (settings.lastCheck == dinozList.length - 1) {
182 				settings.state = STATE.WAIT;
183 				settings.isCheck = true;
184 				settings.lastCheck = 0;
185 				saveSettings();
186 			} else {
187 				saveSettings("lastCheck", settings.lastCheck + 1);
188 			}
189 		}
190 		
191 		redirect("/dino/" + dinozList[settings.lastCheck].id);
192 	}

198 			
199 			settings.state = STATE.GET_TIME;
200 			settings.isCheck = false;
201 			settings.lastCheck = 0;
202 			settings.time = 0;
203 			
204 			saveSettings();
205 			
206 			return false;
207 		} else {
208 			return true;
209 		}
210 	}

298 	/**
299 	 * Возвращает "случайное" время, на основании параметров
300 	 *
301 	 * @params {object} time 
302 	 * @return {int} Время в миллисекундах
303 	 */
304 	var getRandomTime = function (time) {
305 		return Math.round((time.randTime * Math.random() + time.minTime) * time.count);
306 	}
307 	
308 	/**
309 	 * Возвращает строку с временем
310 	 * 
311 	 * @params {int} time Время в миллисекундах
312 	 * @return {string} Время в минутах и секундах
313 	 */
314 	var getTimeStr = function (time) {
315 		time = new Date(time);
316 		return time.getMinutes() + " минут " + time.getSeconds() + " секунд";
317 	}

337 	 * @params {string} text Текст сообщения

347 			}
348 			if (addText) {
349 				console.log(addText);
350 			}
351 			document.location.pathname = page;
