 49

 54 		});

 57 			autoFightSettings.clearTime();
 58 		});

 60 	
 61 	function saveSettings (key, minTime, randTime, countTime) {
 62 		if (!isNaN(minTime) && !isNaN(randTime) && !isNaN(countTime)) {
 63 			autoFightSettings.setSettings(key, {
 64 				"minTime"  : minTime,
 65 				"randTime" : randTime,
 66 				"count"    : countTime
 67 			});
 68 		}
 69 	}
 70 	
 71 	function isLocalStorageAvailable () {
 72 		try {
 73 			return 'localStorage' in window && window['localStorage'] !== null;
 74 		} catch (e) {
 75 			console.log("Ваш браузер не поддерживает Local Storage");
 76 			return false;
 77 		}
 78 	}

 81 /**
 82  * Настройки автофармилки
 83  *
 84  * @class
 85  */
 86 var AutoFightSettings = function () {
 87 	var dinozList = [];
 88 	var settings = {};
 89 	var STATE = {
 90 		GET_TIME : 0,
 91 		WAIT     : 1,
 92 		NEW_TURN : 2,
 93 		FIGHT    : 3,
 94 		SHOP     : 4,
 95 		HEAL     : 5
 96 	}
 97 	
 98 	/** Конструктор */
 99 	var AutoFightSettings = function () {
100 		getDinozList();
101 		updateList();
102 		
103 		getSettings();
104 		showSettings();
105 	}
106 	
107 	/** Получает список динозов */
108 	var getDinozList = function () {
109 		var dinoz = localStorage.getItem("dinoz_list");
110 		if (dinoz == null) {
111 			console.log("Список динозов пуст");
112 		} else {
113 			dinozList = JSON.parse(dinoz);
114 		}
115 	}
116 	
117 	/** Получает настройки */
118 	var getSettings = function () {
119 		var s = localStorage.getItem("settings");
120 		if (s == null) {
121 			settings = getDefaultSettings();
122 			saveSettings();
123 		} else {
124 			settings = JSON.parse(s);
125 		}
126 	}
127 	
128 	/**
129 	 * Добавляет диноза
130 	 * 
131 	 * @param {int} id Айди диноза
132 	 */
133 	this.add = function (id) {
134 		console.log(id, dinozList);
135 		if (id != '') {
136 			dinozList.push({
137 				'id'     : id,
138 				'tCount' : 0,
139 				'hp'     : 0,
140 				'gold'   : 0
141 			})
142 			
143 			settings.state = STATE.GET_TIME;
144 			settings.isCheck = false;
145 			settings.lastCheck = 0;
146 			settings.time = 0;
147 			
148 			saveSettings();
149 			saveDinozList();
150 		}
151 	}
152 	
153 	/**
154 	 * Удаляет диноза
155 	 * 
156 	 * @param {int} id Айди диноза
157 	 */
158 	this.remove = function (id) {
159 		for (var i = 0, length = dinozList.length; i < length; i++) {
160 			if (dinozList[i].id == id) {
161 				dinozList.splice(i, 1);
162 				saveDinozList();
163 				
164 				settings.state = STATE.GET_TIME;
165 				settings.isCheck = false;
166 				settings.lastCheck = 0;
167 				settings.time = 0;
168 				
169 				saveSettings();
170 				
171 				break;
172 			}
173 		}
174 	}
175 	
176 	/** Сохраняет список динозов */
177 	var saveDinozList = function () {
178 		localStorage.setItem("dinoz_list", JSON.stringify(dinozList));
179 		updateList();
180 		console.log(dinozList);
181 	}
182 	
183 	/** Сохраняет настройки */
184 	var saveSettings = function () {
185 		localStorage.setItem("settings", JSON.stringify(settings));
186 	}
187 	
188 	/**
189 	 * Изменяет значение поля настроек
190 	 *
191 	 * @param {string} key Имя поля
192 	 * @param {} value Значение
193 	 */
194 	this.setSettings = function (key, value) {
195 		settings[key] = value;
196 		saveSettings();
197 	}
198 	
199 	/** Удаляет сохранённых динозов*/
200 	this.clearDinoz = function () {
201 		dinozList = [];
202 		saveDinozList();
203 		
204 		settings.state = STATE.GET_TIME;
205 		settings.isCheck = false;
206 		settings.lastCheck = 0;
207 		settings.time = 0;
208 		
209 		saveSettings();
210 		
211 		console.log("Локальное хранилище очищено!");
212 	}
213 	
214 	/** Удаляет все сохранённые данные */
215 	this.clear = function () {
216 		localStorage.clear();
217 		dinozList = [];
218 		settings = getDefaultSettings();
219 		saveSettings();
220 		
221 		updateList();
222 		console.log("Локальное хранилище очищено!");
223 	}
224 	
225 	/** Сбрасывает время нового хода */
226 	this.clearTime = function () {
227 		settings.state = STATE.GET_TIME;
228 		settings.isCheck = false;
229 		settings.lastCheck = 0;
230 		settings.time = 0;
231 		
232 		saveSettings();
233 	}
234 	
235 	/** Выводит массив динозов */
236 	var updateList = function () {
237 		var dinozListEl = $("#dinoz_list tbody");
238 		dinozListEl.empty();
239 		for (var i = 0, length = dinozList.length; i < length; i++) {
240 			dinozListEl.append("<tr><td>" + 
241 									dinozList[i].id + "</td><td>" + 
242 									dinozList[i].tCount + "</td><td>" + 
243 									dinozList[i].hp + "</td><td>" + 
244 									dinozList[i].gold + "</td></tr>"   /*+
245 									"<td class='remove_dinoz' onclick='self.remove(" + dinozList[i].id + ")'></td></tr>"*/
246 			);
247 		}
248 	}
249 	
250 	/** Выводит настройки на форму */
251 	var showSettings = function () {
252 		$("#redirect .min_time:first").val(settings.redirectTime.minTime); 
253 		$("#redirect .rand_time:first").val(settings.redirectTime.randTime);  
254 		$("#redirect .count:first").val(settings.redirectTime.count); 
255 		
256 		$("#new_turn .min_time:first").val(settings.newTurnTime.minTime); 
257 		$("#new_turn .rand_time:first").val(settings.newTurnTime.randTime);
258 		$("#new_turn .count:first").val(settings.newTurnTime.count);
259 		
260 		$("#fight .min_time:first").val(settings.fightTime.minTime); 
261 		$("#fight .rand_time:first").val(settings.fightTime.randTime);
262 		$("#fight .count:first").val(settings.fightTime.count);
263 		
264 		$("#continue .min_time:first").val(settings.continueTime.minTime); 
265 		$("#continue .rand_time:first").val(settings.continueTime.randTime);
266 		$("#continue .count:first").val(settings.continueTime.count);
267 	}
268 	
269 	/**
270 	 * Возвращает дефолтные настройки
271 	 * 
272 	 * @return {Object} дефолтные настройки
273 	 */
274 	var getDefaultSettings = function () {
275 		return {
276 			isActive   : false,
277 			isCheck    : false,
278 			lastCheck  : 0,
279 			time       : 0,
280 			state      : STATE.GET_TIME,
281 			redirectTime : {
282 				minTime  : 3,
283 				randTime : 2,
284 				count    : 1000
285 			},
286 			newTurnTime  : {
287 				minTime  : 1,
288 				randTime : 9,
289 				count    : 60000
290 			},
291 			fightTime    : {
292 				minTime  : 1.5,
293 				randTime : 2,
294 				count    : 1000
295 			},
296 			continueTime : {
297 				minTime  : 3,
298 				randTime : 2,
299 				count    : 1000
300 			}
301 		};
302 	}
303 	
304 	AutoFightSettings();
305 }
